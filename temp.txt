class CreateStudent(BaseModel):
    firstname:str
    lastname:str
    matric_no:str

    @validator("matric_no")
    def validate_matric_no(cls, v):

        try:
            if len(v) != 11: raise BadRequestException("matric no must be lenght of 10")

            return int(v)
        
        except ValueError as e:
            raise BadRequestException(str(e))
